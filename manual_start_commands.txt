Start backend (terminal 1):
    cd backend
    If it is the first time:
        python3 -m venv venv
        source venv/bin/activate
        pip install -r requirements.txt
    else:
        source venv/bin/activate
    source .env
    uvicorn app.main:app --host 0.0.0.0 --port 8000 --ssl-certfile "$BACKEND_CERT_PATH" --ssl-keyfile "$BACKEND_KEY_PATH" --ssl-keyfile-password "$SSL_KEY_PASSPHRASE"

Start frontend (terminal 2)
    cd frontend
    If first time
        npm install
    npm run dev -- --host
